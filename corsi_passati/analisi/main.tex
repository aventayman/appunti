\chapter{Ripasso}
\dfn{Estremo superiore per un insieme}{Sia \(A \neq \emptyset\) un insieme allora chiamiamo \(\sup A = L\) l'estremo superiore di \(A\) se
\begin{itemize}
    \item \(\forall x \in A \quad L \ge x\) 
    \item \(\forall \varepsilon > 0 \exists x \in A : \quad L - \varepsilon < x\)
\end{itemize}}
\dfn{Estremo inferiore per un insieme}{Sia \(A \neq \emptyset\) un insieme, allora chiamiamo \(\inf A = I\) l'estremo inferiore di \(A\) se
\begin{itemize}
    \item \(\forall x \in A \quad I \le x\) 
    \item \(\forall \varepsilon > 0 \exists x \in A : \quad I + \varepsilon > x\)
\end{itemize}}
